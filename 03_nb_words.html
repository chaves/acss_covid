def get_nb_words(text):
    text = text.strip() # remove leading and trailing whitespaces
    text = text.translate(str.maketrans('', '', string.punctuation)) # punctuation
    words = text.split() # make list
    words = [word.replace("\n", "") for word in words] # no lines
    words = [word for word in words  if not word.isdigit()] # no digits
    words = [word for word in words if len(word) > 1] # only words with more than one letter
    return len(words) # nb of words
